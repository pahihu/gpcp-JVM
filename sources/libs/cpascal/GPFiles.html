MODULE GPFiles;

IMPORT
    RTS;

TYPE

  FILE* = POINTER TO ABSTRACT RECORD
           (* class | no noArg-ctor | no value-assign *)
          END;

  FileNameArray* = POINTER TO ARRAY OF CHAR;


VAR
  fileSep- : CHAR;
  pathSep- : CHAR;
  optChar- : CHAR;


PROCEDURE CurrentDirectory*() : FileNameArray;

PROCEDURE isOlder*(first  : FILE;
                   second : FILE) : BOOLEAN;

PROCEDURE MakeDirectory*(dirName : ARRAY OF CHAR);

PROCEDURE FileList*(IN  dirPath : ARRAY OF CHAR) : POINTER TO ARRAY OF FileNameArray;

PROCEDURE exists*(fName : ARRAY OF CHAR) : BOOLEAN;


END GPFiles.
